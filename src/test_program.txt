CONSTANTS
    =, +, *, 0
AXIOMS
    (= a a)
    (= a b) => (= b a)
    (= a b) and (= b c) => (= a c)

    (= (+ a b) (+ b a))
    (= (+ a (+ b c)) (+ (+ a b) c))
    (= a b) <=> (= (+ a c) (+ b c))

    (= (* a b) (* b a))
    (= (* a (* b c)) (* (* a b) c))
    (= a b) <=> (= (* a c) (* b c))
    (= (* a (+ b c)) (+ (* a b) (* a c)))
THEOREM
    (= (* x (+ x 0)) (+ (* x (+ x 0)) 0))
    (= (* x (+ x 0)) (+ (* x x) 0))
    (= (* x (+ x 0)) (+ (* x x) (* x 0)))
    (= (+ (* x x) 0) (+ (* x x) (* x 0)))
    (= 0 (* x 0))
